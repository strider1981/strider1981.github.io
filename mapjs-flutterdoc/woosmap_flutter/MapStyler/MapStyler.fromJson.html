MapStyler.fromJson(Map<String, dynamic> json) {
  if (json['color'] != null) {
    color = json['color'].toString();
  }
  if (json['gamma'] != null) {
    color = json['gamma'].toDouble();
  }
  if (json['hue'] != null) {
    hue = json['hue'].toString();
  }
  if (json['invert_lightness'] != null) {
    invertLightness = json['invert_lightness'] as bool;
  }
  if (json['lightness'] != null) {
    lightness = json['lightness'].toDouble();
  }
  if (json['saturation'] != null) {
    saturation = json['saturation'].toDouble();
  }
  if (json['visibility'] != null) {
    visibility = json['visibility'].toString();
  }
  if (json['weight'] != null) {
    weight = json['weight'].toDouble();
  }
}
