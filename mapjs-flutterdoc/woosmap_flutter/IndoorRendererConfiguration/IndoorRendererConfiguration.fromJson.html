IndoorRendererConfiguration.fromJson(Map<String, dynamic> json) {
  if (json['centerMap'] != null) {
    centerMap = json['centerMap'] as bool;
  }

  if (json['venue'] != null) {
    venue = json['venue'].toString();
  }

  if (json['defaultFloor'] != null) {
    defaultFloor = json['defaultFloor'] as int;
  }

  if (json['highlightPOIByRef'] != null) {
    highlightPOIByRef = json['highlightPOIByRef'].toString();
  }

  if (json['icon'] != null) {
    icon = WoosIcon.fromJson(json['highlightPOIByRef']);
  }

  if (json['disablePOIInfowindow'] != null) {
    disablePOIInfowindow = json['disablePOIInfowindow'] as bool;
  }

  if (json['forceExtrusion'] != null) {
    forceExtrusion = json['forceExtrusion'] as bool;
  }

  if (json['hideLevelSelector'] != null) {
    hideLevelSelector = json['hideLevelSelector'] as bool;
  }

  if (json['useInfoWindow'] != null) {
    useInfoWindow = json['useInfoWindow'] as bool;
  }

  if (json['showRoutingPaths'] != null) {
    showRoutingPaths = json['showRoutingPaths'] as bool;
  }

  if (json['baseFilter'] != null) {
    baseFilter = json['baseFilter'].toString();
  }

  if (json['responsive'] != null) {
    if (json['responsive'].toString() == 'desktop'){
      responsive = IndoorWidgetResponsiveMode.desktop;
    } else if (json['responsive'].toString() == 'mobile'){
      responsive = IndoorWidgetResponsiveMode.mobile;
    }
  }

  if (json['theme'] != null) {
    theme = json['theme'].toString();
  }

  if (json['directionsIcons'] != null) {
    directionsIcons = WoosIndoorDirectionsIcons.fromJson(json['directionsIcons']);
  }

  if (json['levelSelectorPosition'] != null) {
    if (json['levelSelectorPosition'].toString() == 'right'){
      levelSelectorPosition = IndoorLevelSelectorPosition.right;
    } else if (json['levelSelectorPosition'].toString() == 'top'){
      levelSelectorPosition = IndoorLevelSelectorPosition.top;
    }
  }
}
