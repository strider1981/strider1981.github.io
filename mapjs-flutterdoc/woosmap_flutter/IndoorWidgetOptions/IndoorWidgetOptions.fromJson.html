IndoorWidgetOptions.fromJson(Map<String, dynamic> json) {
  bool isPropertySet = false;
  if (json['ui'] != null) {
    if (json['ui'] is Map<String, dynamic>){
      ui = IndoorWidgetOptionsUI.fromJson(json['ui']);
      isPropertySet = true;
    }
    else{
      Util.showWarning("`ui` should be an object of Map<String, dynamic>");
    }
  }

  if (json['units'] != null) {
    if (json['units'].toString() == 'imperial'){
      units = UnitSystem.imperial;
      isPropertySet = true;
    }
  }

  if (json['baseFilter'] != null) {
    baseFilter = json['baseFilter'].toString();
    isPropertySet = true;
  }

  if (json['directionsMode'] != null) {
    directionsMode = json['directionsMode'].toString();
    isPropertySet = true;
  }

  if (json['avoidZones'] != null) {
    avoidZones = json['avoidZones'].toString();
    isPropertySet = true;
  }

  if (json['autocompleteWithDistance'] != null) {
    autocompleteWithDistance = json['autocompleteWithDistance'] as bool;
    isPropertySet = true;
  }

  if (json['navigationMode'] != null) {
    navigationMode = json['navigationMode'] as bool;
    isPropertySet = true;
  }

  if (!isPropertySet){
    Util.showWarning("No valid properties found in given map for IndoorWidgetOptions.");
  }
}
