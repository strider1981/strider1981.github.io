Future<void> filterPois({required String advancedFilter}) {
  //assert(advancedFilter.isNotEmpty);
  if (_loadAsWidget) {
    return _webcontroller
        .runJavaScript(_scriptFormat(
            "woosmap.indoorWidget._renderer.filterPois('$advancedFilter')"))
        .catchError((error) => debugPrint(error));
  } else {
